import pandas as pd
import numpy as np
import random
import time
from collections import Counter

from sklearn.model_selection import KFold
from sklearn.metrics import f1_score, accuracy_score, roc_auc_score
from sklearn.naive_bayes import GaussianNB
from xgboost import XGBClassifier
from pu_learning import spies

SEED = 42
random.seed(SEED)
np.random.seed(SEED)

pd.set_option('display.max_rows', None)
pd.set_option('display.max_columns', None)
pd.set_option('display.width', 1000)
pd.set_option('display.float_format', lambda x: '%.6f' % x)

DATA_PATH = r'C:\Users\Lenovo\Desktop\PRHZ52.CSV'
OUTPUT_PATH = r'C:\Users\Lenovo\Desktop\PU_XGBC_GNB_swapped.csv'
METRICS_PATH = r'C:\Users\Lenovo\Desktop\PU_cv_metrics_swapped.csv'

N_SPLITS = 5
N_BOOTSTRAPS = 80
FINAL_BOOTSTRAPS = 150
UNL_TO_POS_RATIO = 2
THRESHOLD = 0.5
MAX_RETRY_PER_BOOTSTRAP = 30 

def load_data(path):
    texts = pd.read_csv(path, header=None)
    col_label = texts.shape[1] - 52

    labels = dict(Counter(texts[col_label]))
    sorted_labels = sorted(labels.items(), key=lambda x: x[1], reverse=True)

    label_0 = sorted_labels[0][0]   
    label_1 = sorted_labels[1][0]   

    index_1 = [i for i, c in enumerate(texts[col_label]) if c == label_1]
    index_0 = [i for i, c in enumerate(texts[col_label]) if c == label_0]

    texts_1 = texts.iloc[index_1].drop([col_label], axis=1).to_numpy(dtype=float)
    texts_0 = texts.iloc[index_0].drop([col_label], axis=1).to_numpy(dtype=float)

    return texts_1, texts_0

# spies(XGBClassifier, GaussianNB)
# =========================
def build_spies_model(seed=42):
    model = spies(
        XGBClassifier(
            learning_rate=0.05,
            n_estimators=300,
            max_depth=3,
            min_child_weight=2,
            gamma=0.2,
            subsample=0.8,
            colsample_bytree=0.8,
            reg_alpha=0.5,
            reg_lambda=2.0,
            objective='binary:logistic',
            eval_metric='logloss',
            random_state=seed,
            nthread=4
        ),
        GaussianNB()
    )
    return model

def safe_second_model_proba(model, X):
    X = np.array(X)

    sm = model.second_model
    proba = sm.predict_proba(X)
    proba = np.asarray(proba)

    if proba.ndim == 2 and proba.shape[1] >= 2:
        classes = getattr(sm, 'classes_', np.array([0, 1]))
        if 1 in classes:
            idx1 = int(np.where(classes == 1)[0][0])
            return proba[:, idx1].astype(float)
        else:
            return np.zeros(X.shape[0], dtype=float)

    if proba.ndim == 2 and proba.shape[1] == 1:
        classes = getattr(sm, 'classes_', np.array([0]))
        sole_class = classes[0]
        if sole_class == 1:
            return np.ones(X.shape[0], dtype=float)
        else:
            return np.zeros(X.shape[0], dtype=float)

    if proba.ndim == 1:
        return np.clip(proba.astype(float), 0.0, 1.0)

def corrected_prob_1(model, X_target, X_pos_ref, X_unl_ref):
    raw_pos = safe_second_model_proba(model, X_pos_ref)
    raw_unl = safe_second_model_proba(model, X_unl_ref)

    reverse_flag = np.mean(raw_pos) < np.mean(raw_unl)

    raw_target = safe_second_model_proba(model, X_target)
    prob_1 = 1.0 - raw_target if reverse_flag else raw_target
    prob_1 = np.clip(prob_1, 0.0, 1.0)

    return prob_1, reverse_flag, float(np.mean(raw_pos)), float(np.mean(raw_unl))

def second_model_is_valid(model):
    if not hasattr(model, 'second_model'):
        return False

    sm = model.second_model
    classes = getattr(sm, 'classes_', None)
    if classes is None:
        return False

    return len(classes) >= 2

def fit_one_balanced_spies(X_pos_train, X_unl_pool, seed):
    n_pos = len(X_pos_train)

    for retry in range(MAX_RETRY_PER_BOOTSTRAP):
        rng = np.random.RandomState(seed + retry)

        n_unl_sample = min(len(X_unl_pool), max(n_pos * UNL_TO_POS_RATIO, n_pos))
        sampled_idx = rng.choice(len(X_unl_pool), size=n_unl_sample, replace=False)
        X_unl_sample = X_unl_pool[sampled_idx]

        X_train = pd.DataFrame(np.vstack([X_pos_train, X_unl_sample]))
        y_train = pd.Series([1] * len(X_pos_train) + [0] * len(X_unl_sample))

        model = build_spies_model(seed + retry)
        model.fit(X_train, y_train)

        if second_model_is_valid(model):
            return model, X_unl_sample, retry

def main():
    start = time.time()

    texts_1, texts_0 = load_data(DATA_PATH)

    n_pos = len(texts_1)
    n_unl = len(texts_0)

    n_splits = min(N_SPLITS, n_pos)

    kf = KFold(n_splits=n_splits, shuffle=True, random_state=SEED)

    pos_oof_prob_1 = np.zeros(n_pos)
    pos_oof_count = np.zeros(n_pos)
    unl_fold_probs = []
    metrics_records = []

    for fold_id, (train_idx, valid_idx) in enumerate(kf.split(texts_1), start=1):
        print(f"\n========== Fold {fold_id} ==========")

        p_train = texts_1[train_idx]
        p_valid = texts_1[valid_idx]

        prob_pos_valid_sum = np.zeros(len(p_valid))
        prob_unl_all_sum = np.zeros(n_unl)

        rng_eval = np.random.RandomState(SEED + 10000 + fold_id)
        eval_unl_idx = rng_eval.choice(n_unl, size=len(p_valid), replace=False)
        u_eval = texts_0[eval_unl_idx]
        prob_u_eval_sum = np.zeros(len(u_eval))

        reverse_flags = []
        retry_used_list = []

        for b in range(N_BOOTSTRAPS):
            seed_b = SEED + fold_id * 1000 + b

            model, u_train_sub, retry_used = fit_one_balanced_spies(
                X_pos_train=p_train,
                X_unl_pool=texts_0,
                seed=seed_b
            )
            retry_used_list.append(retry_used)

            prob_pos_valid, reverse_flag, raw_pos_mean, raw_unl_mean = corrected_prob_1(
                model, p_valid, p_train, u_train_sub
            )
            prob_unl_all, _, _, _ = corrected_prob_1(
                model, texts_0, p_train, u_train_sub
            )
            prob_u_eval, _, _, _ = corrected_prob_1(
                model, u_eval, p_train, u_train_sub
            )

            prob_pos_valid_sum += prob_pos_valid
            prob_unl_all_sum += prob_unl_all
            prob_u_eval_sum += prob_u_eval
            reverse_flags.append(int(reverse_flag))

        prob_pos_valid_mean = prob_pos_valid_sum / N_BOOTSTRAPS
        prob_unl_all_mean = prob_unl_all_sum / N_BOOTSTRAPS
        prob_u_eval_mean = prob_u_eval_sum / N_BOOTSTRAPS

        pos_oof_prob_1[valid_idx] += prob_pos_valid_mean
        pos_oof_count[valid_idx] += 1
        unl_fold_probs.append(prob_unl_all_mean)

        y_eval = np.array([1] * len(p_valid) + [0] * len(u_eval))
        prob_eval = np.concatenate([prob_pos_valid_mean, prob_u_eval_mean])
        pred_eval = (prob_eval >= THRESHOLD).astype(int)

        f1 = f1_score(y_eval, pred_eval)
        acc = accuracy_score(y_eval, pred_eval)
        auc = roc_auc_score(y_eval, prob_eval)

        metrics_records.append({
            'Fold': fold_id,
            'Reversed_ratio': np.mean(reverse_flags),
            'Mean_retry_used': np.mean(retry_used_list),
            'F1': f1,
            'Accuracy': acc,
            'AUC_ROC': auc,
            'Pos_mean_probability_1': float(np.mean(prob_pos_valid_mean)),
            'Unl_mean_probability_1': float(np.mean(prob_u_eval_mean))
        })

        print(f"Fold {fold_id} F1: {f1:.6f}")
        print(f"Fold {fold_id} Accuracy: {acc:.6f}")
        print(f"Fold {fold_id} AUC-ROC: {auc:.6f}")

    pos_prob_oof_1 = pos_oof_prob_1 / np.maximum(pos_oof_count, 1)
    unl_prob_cv_1 = np.mean(np.vstack(unl_fold_probs), axis=0)

    print("\n========== Final Training ==========")

    final_pos_sum = np.zeros(n_pos)
    final_unl_sum = np.zeros(n_unl)
    final_reverse_flags = []
    final_retry_list = []

    for b in range(FINAL_BOOTSTRAPS):
        seed_b = SEED + 50000 + b

        model, u_train_sub, retry_used = fit_one_balanced_spies(
            X_pos_train=texts_1,
            X_unl_pool=texts_0,
            seed=seed_b
        )
        final_retry_list.append(retry_used)

        prob_pos_final, reverse_flag, _, _ = corrected_prob_1(
            model, texts_1, texts_1, u_train_sub
        )
        prob_unl_final, _, _, _ = corrected_prob_1(
            model, texts_0, texts_1, u_train_sub
        )

        final_pos_sum += prob_pos_final
        final_unl_sum += prob_unl_final
        final_reverse_flags.append(int(reverse_flag))

    final_pos_prob_1 = final_pos_sum / FINAL_BOOTSTRAPS
    final_unl_prob_1 = final_unl_sum / FINAL_BOOTSTRAPS

    pos_prob_output_1 = 0.7 * pos_prob_oof_1 + 0.3 * final_pos_prob_1
    unl_prob_output_1 = 0.5 * unl_prob_cv_1 + 0.5 * final_unl_prob_1

    pos_prob_output_1 = np.clip(pos_prob_output_1, 0.0, 1.0)
    unl_prob_output_1 = np.clip(unl_prob_output_1, 0.0, 1.0)

    pos_prob_output_0 = 1.0 - pos_prob_output_1
    unl_prob_output_0 = 1.0 - unl_prob_output_1

    df_pos_out = pd.DataFrame(texts_1)
    df_pos_out['sample_type'] = 'positive_1'
    df_pos_out['probability_0'] = pos_prob_output_0
    df_pos_out['probability_1'] = pos_prob_output_1

    df_unl_out = pd.DataFrame(texts_0)
    df_unl_out['sample_type'] = 'unlabeled_0'
    df_unl_out['probability_0'] = unl_prob_output_0
    df_unl_out['probability_1'] = unl_prob_output_1

    df_out = pd.concat([df_pos_out, df_unl_out], axis=0, ignore_index=True)

    cols = list(df_out.columns)
    fixed_cols = [c for c in cols if c not in ['probability_0', 'probability_1']]
    df_out = df_out[fixed_cols + ['probability_0', 'probability_1']]

    df_out.to_csv(OUTPUT_PATH, index=False, encoding='utf-8-sig')

    metrics_df = pd.DataFrame(metrics_records)
    mean_row = pd.DataFrame([{
        'Fold': 'Mean',
        'Reversed_ratio': metrics_df['Reversed_ratio'].mean(),
        'Mean_retry_used': metrics_df['Mean_retry_used'].mean(),
        'F1': metrics_df['F1'].mean(),
        'Accuracy': metrics_df['Accuracy'].mean(),
        'AUC_ROC': metrics_df['AUC_ROC'].mean(),
        'Pos_mean_probability_1': metrics_df['Pos_mean_probability_1'].mean(),
        'Unl_mean_probability_1': metrics_df['Unl_mean_probability_1'].mean()
    }])
    metrics_out = pd.concat([metrics_df, mean_row], ignore_index=True)
    metrics_out.to_csv(METRICS_PATH, index=False, encoding='utf-8-sig')

if __name__ == '__main__':
    main()
